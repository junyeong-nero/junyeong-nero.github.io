## 1020. Number of Enclaves (medium)

```python
class Solution:
    def numEnclaves(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        def dfs(x, y):
            if x < 0 or x >= m or y < 0 or y >= n:
                return
            if grid[x][y] == 0:
                return
            grid[x][y] = 0
            dirs = [1, 0, -1, 0, 1]
            for i in range(4):
                dfs(x + dirs[i], y + dirs[i + 1])
        
        for i in range(m):
            dfs(i, 0)
            dfs(i, n - 1)
        
        for j in range(n):
            dfs(0, j)
            dfs(m - 1, j)
        
        print(grid)
        count = 0
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 1:
                    # dfs(i, j)
                    count += 1
        return count
```
