## 2336. Smallest Number in Infinite Set (medium)

```python
class SmallestInfiniteSet:

    def __init__(self):
        self.d = defaultdict(int)
        self.small = 1


    def popSmallest(self) -> int:
        cur = self.small
        self.d[self.small] -= 1
        while self.d[self.small] == -1:
            self.small += 1
        return cur

    def addBack(self, num: int) -> None:
        if num < self.small:
            self.small = num
        self.d[num] = 0
```
