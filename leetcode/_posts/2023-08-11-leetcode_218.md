## 518. Coin Change II (medium)

weird DP, my solution

```python
class Solution:
    def change(self, amount: int, coins: List[int]) -> int:
        n = len(coins)
        dp = defaultdict(list)
        dp[0] = [0] * n
        dp[0][0] = 1
        
        for cur in range(1, amount + 1):
            temp = [0] * n
            for i in range(n): 
                arr = dp[cur - coins[i]]
                if cur - coins[i] == 0:
                    arr = [0] * n
                    arr[i] = 1
                if not arr:
                    continue
                for j in range(i + 1):
                    temp[i] += arr[j]
            dp[cur] = temp
        
        # print(dp)

        return sum(dp[amount])
```
