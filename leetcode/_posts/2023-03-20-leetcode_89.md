## 605. Can Place Flowers (easy)

```python
class Solution:
    def canPlaceFlowers(self, flowerbed: List[int], n: int) -> bool:
        m = len(flowerbed)
        count = 0
        for i in range(m):
            a = b = c = False
            if i - 1 < 0 or flowerbed[i - 1] == 0: a = True
            if i + 1 >= m or flowerbed[i + 1] == 0: b = True
            if flowerbed[i] == 0: c = True
            if a and b and c:
                flowerbed[i] = 1
                count += 1

        # print(flowerbed, count)
        return n <= count
```

## 326. Power of Three (easy)

```python
class Solution:
    def isPowerOfThree(self, n: int) -> bool:
        if n <= 0: return False
        if n == 1: return True
        if n % 3 != 0: return False
        return self.isPowerOfThree(n // 3)
```

## 331. Verify Preorder Serialization of a Binary Tree (medium)

```python
class Solution:
    def isValidSerialization(self, preorder: str) -> bool:
        split = preorder.split(',')
        def dfs(arr):
            # print(arr)
            if arr[0] == '#':
                return 1
            if len(arr) < 3:
                return 0
            if arr[1] == '#' and arr[2] == '#':
                # leaf node, next node is behind 3 index
                return 3
            left = dfs(arr[1:])
            right = dfs(arr[1 + left:])
            return 1 + left + right
        
        return dfs(split) == len(split)
```
