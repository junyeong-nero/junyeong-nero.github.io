## 1498. Number of Subsequences That Satisfy the Given Sum Condition (medium)

TLE at 55/62

```python
class Solution:
    def numSubseq(self, nums: List[int], target: int) -> int:

        n = len(nums)
        nums = sorted(nums)

        def help(index, k):
            p = target - k
            if p < k:
                return 0
            count = 0
            for i in range(index + 1, n):
                temp = nums[i]
                if i == index:
                    continue
                if temp >= k and temp <= p:
                    count += 1
            return 2 ** count

        res = 0
        for index, num in enumerate(nums):
            a = help(index, num)
            # print(index, num, a)
            res += a
            res %= 10 ** 9 + 7

        return res
            
```
