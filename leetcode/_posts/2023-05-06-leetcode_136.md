## 1498. Number of Subsequences That Satisfy the Given Sum Condition (medium)

TLE at 57/62

```python
class Solution:
    def numSubseq(self, nums: List[int], target: int) -> int:

        n = len(nums)
        nums = sorted(nums)

        d = {}

        def help(index, k):
            p = target - k
            if k in d:
                d[k] //= 2
                return d[k]
            if p < k:
                return 0
            count = 0
            for i in range(index + 1, n):
                temp = nums[i]
                if i == index:
                    continue
                if temp >= k and temp <= p:
                    count += 1
                else:
                    break
            d[k] = 2 ** count
            return d[k]

        res = 0
        for index, num in enumerate(nums):
            res += help(index, num)
            res %= 10 ** 9 + 7

        return res

class Solution:
    def numSubseq(self, nums: List[int], target: int) -> int:
        res, mod = 0, 1000000007
        l, r = 0, len(nums) - 1
        pre = [1]
        for i in range(1, len(nums) + 1):
            pre.append((pre[-1] << 1) % mod)
                
        nums.sort()
        
        while l <= r:
            if nums[l] + nums[r] > target:
                r -= 1
            else:
                res = (res + pre[r - l]) % mod
                l += 1

        return res
```

## 1558. Minimum Numbers of Function Calls to Make Target Array (medium)

class Solution:
    def minOperations(self, nums: List[int]) -> int:
        def isAllEven(arr):
            for num in arr:
                if num % 2 == 1:
                    return False
            return True
        
        n = len(nums)
        end = [0] * n

        res = 0
        while sum(nums) != 0:

                for index, num in enumerate(nums):
                    nums[index] //= 2
                res += 1
            else:
